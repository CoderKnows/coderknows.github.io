Бэкап и восстановление

zcat /path/to/file.sql.gz | mysql -u USERNAME -pPASSWORD your_database
восстанавливает базу данных из файла в архиве gz
tar -xzOf /path/to/file.sql.tar.gz | mysql -u USERNAME -pPASSWORD your_database
восстанавливает базу данных из файла в архиве tar.gz
